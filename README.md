    "tan": tangent,
    "cot": cotangent,
    "wir": weierstrass,
    "esin": esin,
    "sin": sine,
    "tri": triangle,
    "saw": sawtooth,
    "sqr": square,
    "pls": pulse,
    "semi": semicircle,
    "nsin": nestedsine,
    "clx": clx,
    "slx": slx,
    "msaw": msaw,
    "skew": skewer,
    "gsin": gaussiansine,
    "lsin": logsin,
    "art": artifact,
    "osin": orderedsine,
    "alog": antilogarithm,
    "erfs": errorsine,
    "tetra": tetration,
    "decr": decreasingfrequency,
    "rsaw": randomsaw,
